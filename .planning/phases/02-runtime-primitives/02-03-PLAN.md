```
# _fopen(path: l, mode: l): l — open file, return FILE* as opaque 64-bit pointer
export function l $_fopen(l %path, l %mode) {
@start
    %fp =l call $fopen(l %path, l %mode)
    ret %fp
}

# _fclose(fp: l): w — close file, return 0 on success
export function w $_fclose(l %fp) {
@start
    %r =w call $fclose(l %fp)
    ret %r
}
```

**Part B: Add C builtins for _fread_all and _fwrite_str (per D-01)**

These need malloc/complex logic — go in `builtin/builtin_qbe.c`. Append:

```c
/* _fread_all(fp) — read entire file into a malloc'd string, return it */
char *_fread_all(FILE *fp)
{
    size_t capacity = 4096;
    size_t length = 0;
    char *buf = malloc(capacity);
    if (!buf) return "";
    while (1) {
        size_t n = fread(buf + length, 1, capacity - length - 1, fp);
        if (n == 0) break;
        length += n;
        if (length + 1 >= capacity) {
            capacity *= 2;
            buf = realloc(buf, capacity);
            if (!buf) return "";
        }
    }
    buf[length] = '\0';
    return buf;
}

/* _fwrite_str(fp, s) — write string s to file fp, return bytes written */
long _fwrite_str(FILE *fp, char *s)
{
    size_t len = strlen(s);
    size_t written = fwrite(s, 1, len, fp);
    fflush(fp);
    return (long)written;
}
```

**Part C: Add return types to infer_builtin**

In `src/parser/infer.rs` `infer_builtin`, add:
- `"_fopen"` => `Some(Type::Str)` — FILE* is a 64-bit pointer, must use string/long to avoid truncation. Antimony's `string` maps to `qbe::Type::Long` (64-bit), while `int` maps to `qbe::Type::Word` (32-bit). Storing a 64-bit FILE* in a 32-bit int would silently truncate the upper bits, producing an invalid pointer.
- `"_fclose"` => `Some(Type::Int)` — return status code (small integer, word-safe)
- `"_fread_all"` => `Some(Type::Str)` — returns malloc'd string
- `"_fwrite_str"` => `Some(Type::Int)` — returns bytes written

**Part D: Add Antimony wrappers to lib/io.sb**

Check existing `lib/io.sb` contents first. Add these wrapper functions (do not remove existing content).

IMPORTANT: The file handle type is `string`, not `int`. The `string` type maps to `qbe::Type::Long` (64-bit), which correctly holds the 64-bit `FILE*` pointer returned by `fopen`. Using `int` (`qbe::Type::Word`, 32-bit) would silently truncate the pointer on 64-bit systems.

```
// Opens a file at the given path with the given mode ("r", "w", "a", etc.)
// Returns an opaque file handle (string holds 64-bit FILE* pointer). Returns 0 if open fails.
fn file_open(path: string, mode: string): string {
    return _fopen(path, mode)
}

// Reads the entire contents of an open file handle into a string
fn file_read(handle: string): string {
    return _fread_all(handle)
}

// Writes a string to an open file handle. Returns bytes written.
fn file_write(handle: string, content: string): int {
    return _fwrite_str(handle, content)
}

// Closes an open file handle. Returns 0 on success.
fn file_close(handle: string): int {
    return _fclose(handle)
}
```

Append to `RUNTIME_PREAMBLE` in `src/generator/qbe.rs`:

```
# Global storage for argc/argv (stashed at main entry)
data $__argc = { w 0 }
data $__argv = { l 0 }

# _argc(): w — return the number of CLI arguments
export function w $_argc() {
@start
    %n =w loadw $__argc
    ret %n
}

# _argv(i: w): l — return the i-th CLI argument as a C string pointer
export function l $_argv(w %i) {
@start
    %base =l loadl $__argv
    %offset =l extsw %i
    %scaled =l mul %offset, 8
    %addr =l add %base, %scaled
    %ptr =l loadl %addr
    ret %ptr
}
```
